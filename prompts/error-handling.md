```javascript
  try {
    file = fs.openSync(path, 'r');
    const data = fs.readFileSync(file);
    processData(data);
  } catch (error) {
    logger.error('Failed to process file', { path, error });
    throw new FileProcessingError('File processing failed', { cause: error });
  } finally {
    if (file !== undefined) {
      try {
        fs.closeSync(file);
      } catch (closeError) {
        logger.error('Failed to close file', { path, error: closeError });
      }
    }
  }
}
```

## 🔄 Error Recovery Strategies

### Retry Logic

**When to Retry**:
- Network errors
- Temporary service unavailability
- Rate limiting (with backoff)
- Database deadlocks

**When NOT to Retry**:
- Validation errors
- Authentication errors
- Not found errors
- Business logic errors

**Implementation**:
```javascript
async function retryOperation(operation, maxRetries = 3, delay = 1000) {
  for (let attempt = 1; attempt <= maxRetries; attempt++) {
    try {
      return await operation();
    } catch (error) {
      const isLastAttempt = attempt === maxRetries;
      const isRetryable = error instanceof NetworkError || 
                          error instanceof TimeoutError;
      
      if (!isRetryable || isLastAttempt) {
        throw error;
      }
      
      logger.warn('Operation failed, retrying', { 
        attempt, 
        maxRetries, 
        error 
      });
      
      // Exponential backoff
      await sleep(delay * Math.pow(2, attempt - 1));
    }
  }
}
```

### Circuit Breaker Pattern

```javascript
class CircuitBreaker {
  constructor(threshold = 5, timeout = 60000) {
    this.failureCount = 0;
    this.threshold = threshold;
    this.timeout = timeout;
    this.state = 'CLOSED'; // CLOSED, OPEN, HALF_OPEN
    this.nextAttempt = Date.now();
  }
  
  async execute(operation) {
    if (this.state === 'OPEN') {
      if (Date.now() < this.nextAttempt) {
        throw new CircuitBreakerError('Circuit breaker is OPEN');
      }
      this.state = 'HALF_OPEN';
    }
    
    try {
      const result = await operation();
      this.onSuccess();
      return result;
    } catch (error) {
      this.onFailure();
      throw error;
    }
  }
  
  onSuccess() {
    this.failureCount = 0;
    this.state = 'CLOSED';
  }
  
  onFailure() {
    this.failureCount++;
    if (this.failureCount >= this.threshold) {
      this.state = 'OPEN';
      this.nextAttempt = Date.now() + this.timeout;
      logger.error('Circuit breaker opened', { 
        failureCount: this.failureCount 
      });
    }
  }
}
```

```javascript

  operation: 'createUser',
  userId: user.id,
  error: {
    message: error.message,
    code: error.code,
    stack: error.stack
  },
  context: {
    requestId: req.id,
    userId: req.user?.id,
    timestamp: new Date().toISOString()
  }
});
```

### Log Levels

```javascript

class AppError extends Error {
  constructor(message, options = {}) {
    super(message);
    this.name = this.constructor.name;
    this.code = options.code;
    this.statusCode = options.statusCode || 500;
    this.isOperational = true;
    this.context = options.context;
    Error.captureStackTrace(this, this.constructor);
  }
}

// Specific error types
class ValidationError extends AppError {
  constructor(message, options) {
    super(message, { ...options, statusCode: 400, code: 'VALIDATION_ERROR' });
  }
}

  constructor(message, options) {
    super(message, { ...options, statusCode: 404, code: 'NOT_FOUND' });
  }
}
```
